Welcome the the electron-touchscreen demo

Use the following key commands

  • CMD+K (macOS) or CTRL+K (Windows): Toggle fullscreen mode
  • CMD+C (macOS) or CTRL+C (Windows): Toggle cursor visibility
  • CMD+Q (macOS) or CTRL+Q (Windows): Quit the application

Source code for this app:

	// nodejs
	const path = require('path');
	const url = require('url');

	// npm
	const electron = require('electron');

	// electron-touchscreen
	const TouchscreenWindow = require('../index.js');

	// vars
	let win = null;
	const app = electron.app;

	// application event handlers
	app.on('ready', ()=> {
	  const demo_url = url.format({
	    pathname: path.join(__dirname, 'index.html'),
	    protocol: 'file', 
	    slashes: true,
	  });

	  win = new TouchscreenWindow({url: demo_url});
	});
